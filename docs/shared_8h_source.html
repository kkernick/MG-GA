40 template <typename T = std::vector<std::string>> T static split(const std::string& in, const std::string& delim) {
41 T ret;
42
43 size_t pos = 0, next = 0;
44 while (true) {
45 next = in.find(delim, pos);
46 if (next == std::string::npos) break;
47 ret.emplace_back(in.substr(pos, next - pos));
48 pos = next + 1;
49 }
50 ret.emplace_back(in.substr(pos));
51
52 return ret;
53 }

129 void add(const std::vector<std::string_view>& row, const T& v, const size_t& max, const size_t& x) {
130
131 // If we've reached the end, add the value (No collisions should happen).
132 if (x == max + 1) {
133 if (value != T()) throw std::runtime_error("Tree collision!");
134 value = v;
135 return;
136 }
137
138 // Otherwise, either add the intermediary column, or step into it.
139 const auto& val = row[x];
140 auto find = std::find(states.begin(), states.end(), val);
141 if (find == states.end()) states.emplace_back(val).add(row, v, max, x + 1);
142 else find->add(row, v, max, x + 1);
143 }
144
151 bool in(const std::vector<std::string_view>& row, const size_t& max, const size_t& x) const {
152
153 // If we've finished iteration, make sure something exists.
154 if (x == max + 1) {
155 return value != T();
156 }
157
158 // Otherwise find/add child, and recursively call.
159 const auto& val = row[x];
160 if (states.empty()) return false;
161 auto find = std::find(states.begin(), states.end(), val);
162 if (find == states.end()) return false;
163 return find->in(row, max, x + 1);
164 }
165
166
173 const T& get(const std::vector<std::string_view>& row, const size_t& max, const size_t& x) const {
174
175 // At the end, return the value.
176 if (x == max + 1) {
177 return value;
178 }
179
180 // Find/Add; recursive call.
181 const auto& val = row[x];
182 auto find = std::find(states.begin(), states.end(), val);
183 if (states.empty() || find == states.end()) throw std::runtime_error("Value does not exist");
184 return find->get(row, max, x + 1);
185 }

192 inline bool operator == (const std::string_view& str) const {return key == str;}

212 auto i = root.in(row, (max == SIZE_MAX ? row.size() - 1 : max), 0);
213 if (i) ++hits;

216 }

221 }

246 // Construct a range from a min and max.
247 Range(const size_t& m, const size_t& M) {
248 if (m < M) {min = m; max = M;}
249 else {min = M; max = m;}
250 update_string();
251 }

253 Range(const std::string& s) {
254 assert(s[0] == '[' && s[s.length()-1] == ']');
255 const auto trimmed = s.substr(1, s.length() - 2);
256 const auto minmax = split(trimmed, "-");
257 assert(minmax.size() == 2);
258 min = std::stoi(minmax[0]); max = std::stoi(minmax[1]);
259 out = s;
260 }
261
262 // Get the string.
263 const std::string& str() const {return out;}
264
265 // Check if a integer value exists within the range
266 inline bool in(const size_t& val) const {return val >= min && val <= max;}
267 inline bool in(const Range& val) const {return val.min >= min && val.max <= max;}
268
269 // Get the range of values.
270 inline size_t range() const {return max - min;}
271
272 // Conditional updates to widen bounds.
273 void update_min(const size_t& m) {if (m < min || min == max) min = m;}
274 void update_max(const size_t& M) {if (M > max || min == max) max= M;}
275 void update(const Range& r) {update_min(r.min); update_max(r.max); update_string();}
276 void update_string() {
277 std::stringstream out_stream;
278 out_stream << '[' << min << '-' << max << ']';
279 out = out_stream.str();
280 }

285 inline bool operator < (const Range& r) const {return out < r.out;}

329 static std::string print_title(const std::vector<std::string>& title, const uint64_t& cycle) {
330 std::stringstream ret;
331 std::vector<std::string> lookup = {RED, YELLOW, GREEN, BLUE, VIOLET};
332
333 for (size_t x = 0; x < title.size(); ++x) {
334 for (size_t y = 0; y < title[x].length(); ++y) {
335 size_t h = title[x].length(), v = title.size();
336 switch (mode) {
337 case 0: break;
338 case 1: ret << lookup[(x + y + cycle) % 5]; break;
339 case 2: ret << lookup[(x + cycle) % 5]; break;
340 case 3: ret << lookup[(y + cycle) % 5]; break;
341 case 4: ret << lookup[((v - x) + (h - y) + cycle) % 5]; break;
342 case 5: ret << lookup[((v - x) + cycle) % 5]; break;
343 case 6: ret << lookup[((h - y) + cycle) % 5]; break;
344 default: break;
345 }
346 ret << title[x][y];
347 }
348 ret << END << '\n';
349 }
350 return ret.str();
351 }
