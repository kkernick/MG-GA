1/*
2 * This file contains functions related to domain graphs.
3
4 * Domains are represented as trees, with the area itself, like "Job"
5 * acting as the root node. This identifier must match a column name for
6 * domain data to be embedded into tables.
7 *
8 * A hierarchy is then generated, which has an unbound complexity. For example,
9 * The root node may branch off into "Blue Collar" and "High Collar" which may then
10 * branch into the individual fields, or another classification followed by more fields.
11 * There can be an arbitrary amount of leaves on each tree, with no limit on depth. When
12 * parsing, the tree will be traveressed and generate the path from root to the node,
13 * which gives the program all the possible classifications of a node. For example,
14 * If we wanted to find "Mechanic" we would return Job/Blue Collar/Mechanic, to which
15 * any of these values could technically be used in place (Although we omit the root
16 * node because it's so general we can just suppresss the value entirely).
17 *
18 * Constructing a domain from a file follows the above format, placed into a text file
19 * (The extension is irrelevant), with each line being a node and its relationships.
20 * You need not specify non-leaves, nor domain names themselves, as they will automatically
21 * be generated--look at domains.txt in the example folder.
22 *
23 * The only limitation on this system is that nodes must have unique names in the tree; if
24 * you create two nodes with the name name on different branches, the one found first in
25 * iteration will always be returned (Which makes sense, the getter function only asks for a
26 * node name).
27 *
28 * While this implementation did not reference the MinGen reference, available here:
29 * https://dataprivacylab.org/dataprivacy/projects/kanonymity/kanonymity2.pdf
30 * It uses the same general tree for Domains by coincidence, and because its the most obvious and intuitive
31 * solutions. There are some visualizations of this structure that may be beneficial there if
32 * the code is too dense to parse.
33 */

87 bool find(const std::string_view& child, std::vector<std::string>& stack) const {
88
89 // Go through each child.
90 for (const auto& c : children) {
91
92 // If the child is our node, or it has the node, then emplace that node.
93 // This recursively calls down to the terminal nodes.
94 if (c.value == child || c.find(child, stack)) {
95 stack.emplace_back(c.value);
96 return true;
97 }
98 }
99
100 // If we get here, none of the branches contain the node, so return false.
101 return false;
102 }

115 // Increase the total count of nodes, since we need to emplace.

117 return children.emplace_back(child);

159 size_t breadth(const std::string_view& name) const {
160 // Go through each child, if the node exists, get the total amount of children.
161 for (const auto& c : children) {
162 if (c.value == name) return children.size();
163
164 // Recursively try and find sub-domains.
165 size_t in = c.breadth(name);
166 if (in != 0) return in;
167 }
168 return 0;
169 }

216 static std::vector<Domain> construct(const std::string& filename) {
217 // Our list of domains.
218 std::vector<Domain> ret;
219 if (filename.empty()) return ret;
220
221 // Open the file
222 auto file = std::ifstream(filename);
223 if (!file.is_open()) throw std::runtime_error("Failed to read file");
224 std::string line;
225
226 // A linked list that contains our path
227 std::list<std::string> path;
228
229 // Keep going for each rule
230 while (std::getline(file, line)) {
231
232 // Ignore empty lines
233 if (line.empty()) continue;
234
235 // The syntax is a path separated by /, then a colon to split
236 // the keys that should be put in that path. Such as:
237 // /a/b/c: 1,2,3,4 will create the nodes 1,2,3,4 at node c.
238 auto keypair = shared::split(line, ":");
239 path = shared::split<std::list<std::string>>(keypair[0], "/");
240
241 // Get the root by simply popping the front; with a std::vector, this
242 // would be O(n) complexity. With a list, it's O(1).
243 auto root = path.front();
244 shared::strip(root);
245 path.pop_front();
246
247 // Either find an existing domain, or make one.
248 Domain* d = nullptr;
249 for (auto& x : ret) {
250 if (x.value == root) {d = &x; break;}
251 }
252 if (d == nullptr) {d = &ret.emplace_back(root);}
253
254 // Go through each key that should be added to the path
255 for (auto x : shared::split(keypair[1], ",")) {
256
257 // Strip whitespace, then simply add it to the path, add
258 // it to the domain, and then pop it off the path. Because
259 // we're using a linked list, emplace/pop is O(1)
260 shared::strip(x);
261 path.emplace_back(x);
262 d->add(path);
263 path.pop_back();
264 }
265 }
266 // Return the completed domains.
267 return ret;
268 }
269 };
