120 std::vector<std::string> static mutations(const std::string& value, const column& col, const bool& randomize=false) {
121 std::vector<std::string> ret;
122
123 // Suppress
124 ret.emplace_back("*");
125
126 // Options from the domain
127 if (!col.graph.empty()) {
128 const auto derived = col.graph.find(value);
129 for (const auto& x : derived) ret.emplace_back(x);
130 }
131
132 // THe value itself (Included in the domain, hence the conditional)
133 else if (!value.empty()) {ret.emplace_back(value);}
134
135 // Ranges
136 if (col.t == type::integer) {
137
138 // If the value is already a range, compare it to other ones.
139 if (value[0] == '[') {
140 const auto r = shared::Range(value);
141 for (const auto& range : col.ranges) {
142 if (range.in(r)) ret.emplace_back(range.str());
143 }
144 }
145 else {
146 const auto number = std::stoi(value);
147 for (const auto& range : col.ranges) {
148 if (range.in(number)) ret.emplace_back(range.str());
149 }
150 }
151 }
152
153 // Return
154 if (randomize) std::ranges::shuffle(ret, shared::rng);
155 return ret;
156 }

166 void generate_ranges(column& col) {
167 if (col.t != integer) {
168 col.range = shared::Range(0, col.unique.size());
169 }
170 else {
171 for (const auto& x : col.data) {
172 for (const auto& y : col.data) {
173 if (x == y || x == "*" || y == "*") continue;
174
175 shared::Range range;
176 if (x[0] == '[') range = shared::Range(x);
177 else if (y[0] == '[') range = shared::Range(y);
178 else range = shared::Range(std::stoi(x), std::stoi(y));
179
180 col.ranges.emplace(range);
181 col.range.update(range);
182 }
183 }
184 }
185 col.range.update_string();
186 col.ranges.erase(col.range);
187 }

231 RowIterator(const Table& in, const size_t& r = 0): t(in) {
232 assert(r <= t.rows);

235 // Construct the view vector, and update it.

282 Table(
283 const std::string& filename,
284 const std::vector<domain::Domain>& domains={},
285 std::string delim="",
286 const std::string& types_string="",
287 const std::string& weights_string="",
288 const std::string& s_string=""
289 ) {
290
291 // Open the file
292 auto file = std::ifstream(filename);
293 if (!file.is_open())
294 throw std::runtime_error("Failed to read file");
295 std::string line;
296
297 // Get the header
298 std::getline(file, line);
299
300 // If there is no delimiter, guess.
301 if (delim.empty()) {
302 log("Guessing delimiter. You can use --delim to explicitly provide one.");
303 for (const auto& d : {"\t", " ", ","}) {
304 if (line.find(d) != std::string::npos) {
305 log("Assuming delimiter is: " << (d[0] == '\t' ? "tab" : d));
306 delim = d;
307 break;
308 }
309 }
310 }
311
312 // Split the header, create the map.
313 header = shared::split(line, delim);
314 for (const auto& x : header) {
315 data[x] = column{.width = x.length()};
316 for (const auto& domain : domains) {
317 if (domain.name() == x) {
318 data[x].graph = domain;
319 log("Embedded domain hierarchy for " << x);
320 break;
321 }
322 }
323 }
324
325 // Get the types for each column.
326 auto type_vec = types_string.empty() ? std::vector<std::string>() : shared::split(types_string, ",");
327 if (type_vec.size() < header.size()) {
328 log("Missing types in the provided table are assumed to be strings. You can use --types to explicitly provide them.");
329 while (header.size() >= type_vec.size()) type_vec.emplace_back("s");
330 }
331 else if (type_vec.size() > header.size()) log("Redundant types will be ignored");
332 for (size_t x = 0; x < header.size(); ++x) data[header[x]].t = t_lookup(type_vec[x]);
333
334 // Get weights
335 auto weight_vec = weights_string.empty() ? std::vector<std::string>() : shared::split(weights_string, ",");
336 if (weight_vec.size() < header.size()) {
337 log("Missing weights in the provided table are assumed to be 1.0. You can use --weights to explicitly provide them.");
338 while (header.size() >= weight_vec.size()) weight_vec.emplace_back("1.0");
339 }
340 else if (weight_vec.size() > header.size()) log("Redundant weights will be ignored");
341 for (size_t x = 0; x < header.size(); ++x)
342 data[header[x]].weight = std::stof(weight_vec[x]);
343
344 // Get classification
345 auto class_vec = s_string.empty() ? std::vector<std::string>() : shared::split(s_string, ",");
346 if (class_vec.size() < header.size()) {
347 log("Missing sensitivies in the provided table are assumed to be quasi. You can use --sensitivities to explicitly provide them.");
348 while (header.size() >= class_vec.size()) class_vec.emplace_back("q");
349 }
350 else if (class_vec.size() > header.size()) log("Redundant sensitivities will be ignored");
351 for (size_t x = 0; x < header.size(); ++x)
352 data[header[x]].sensitivity = s_lookup(class_vec[x]);
353
354 // Populate the rest of the table.
355 while (std::getline(file, line)) {
356 auto line_split = shared::split(line, delim);
357 for (size_t x = 0; x < header.size(); ++x) {
358 auto& col = data[header[x]];
359 const auto& name = line_split[x];
360
361 // If there is a graph, validate the data. Note that this is just a warning.
362 // You don't need to have all your data strictly adhere to your domains, it just means
363 // that data cannot be inferred and mutated.
364 if (!col.graph.empty()) {
365 if (col.graph.find(name).empty()) {
366 log("Validation Warning: " << line_split[x] << " does not exist in domain graph: " << col.graph.name());
367 }
368 }
369
370 // Add the data to the vector to preserve the order, then "add" it to the unique set,
371 // with it being discarded if it already exists.
372 col.data.emplace_back(name);
373 col.unique.emplace(name);
374 col.width = std::max(col.width, name.length());
375 }
376 ++rows;
377 }
378
379 // Get ranges
380 for (auto& col : data) generate_ranges(col.second);
381 file.close();
382 }

435 std::cout << "TOTAL: ";
436 if (total == SIZE_MAX) std::cout << "OVERFLOW";

449 Table copy = *this;
