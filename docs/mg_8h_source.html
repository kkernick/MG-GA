1/*
2 * This file contains a heavily optimized MinGen implementation.
3 * The Reference for MinGen:
4 * https://dataprivacylab.org/dataprivacy/projects/kanonymity/kanonymity2.pdf
5 * outlines the idea behind the algorithm, but you almost certainly do not need to read
6 * it. It's far more interested in the mathematics of k-anonymity, precision, and MinGen
7 * then the actual implementation, but Figure 5 outlines the algorithm. These
8 * are the steps:
9 * 1. If the table already meets the k-anonymity requirement, return it.
10 * 2. If not, generate every single possible permutation of the table.
11 * 3. Return the one that meets the k-anonymity requirement with the lowest score.
12 * The take away is that this just a brute-force algorithm, trying every single possible
13 * combination and returning the best one; as the paper admits: this isn't a very
14 * efficient process.
15 *
16 * Rather than simply implementing the algorithm directly, I elected to make an optimized
17 * version of it. This problem, where we enumerate multiple possible states, and use
18 * a scoring function to determine its fitness, is quite similar to algorithms for games
19 * like Chess and Go, where we have an unfathomable search-space of all possible combinations
20 * of a game board, and must determine the best branch to go down. In light of this, we can
21 * use a modified version of Alpha-Beta pruning to dramatically accelerate search speed,
22 * and unlike Chess or Go, can actually search the entire space in a reasonable time for
23 * small tables. Alpha-Beta relies on two players, and keeps track of a best score and worst
24 * score in the current stage of the search space, to which it can prune branches that will
25 * never be taken if both players act perfectly. Likewise, we have a score that we keep
26 * track of, and can make our own pruning, although it's more an Alpha pruning since there's
27 * nobody else here.
28 *
29 * After each column, we check both the k-anonymity and score of the current column, and all
30 * modified columns previously, against the original column. We check to see: if we cut the
31 * table off here, what would the k-anonymity value be? If this value is greater than our
32 * threshold, that means (assuming k=2) that we can uniquely identify one of the modified
33 * rows using only the information within the modified columns (Like if all names are unique,
34 * and we didn't suppress the name, then we can uniquely identify each row solely off the name).
35 * If we can uniquely identify a row using only a subset of the modifications, then no additional
36 * modifications will be able to change this, and we can safely prune this branch in the search space.
37 * Likewise, both our scoring metrics strictly INCREASE score, meaning that the best you can score is
38 * 0 if nothing changed. If the score of the subset is GREATER than our current best value, then
39 * it is impossible for any further modification to lead to it being better (Or even equal) due to the
40 * nature of the scoring function, and we can again prune it.
41 *
42 * These optimizations combine with heavy caching with both scoring functions, and can exhaustively cover
43 * a search space of billions in a couple of seconds. Look at metrics.h for more details about that.
44 */

61 // Our MinGen class.
62 class MinGen {
63 private:
64
65 // The working copy that is changed, and the original that remains constant.
66 // There are two different way we could go about going through each possible state of the Table:
67 // 1. Recursively call a function on new table instances with our modifications.
68 // 2. Keep a single working copy, and make changes, then remove them.
69 //
70 // The former option is cleaner to implement (But not by much), but suffers from catastrophic performance
71 // Due to how many copies are made. With our tables containing a lot of auxiliary information, we can't afford
72 // to make that many copies, so just keep a single copy, and modify it iteratively.
73 table::Table working;
74 const table::Table original;
75
76 // K score and metric
77 size_t k = 2;
78 metric::metric m = metric::metric::md;
79
80 // An upper bounds on iterations, current states reached
81 size_t max = SIZE_MAX, states = 0;
82
83 // A list of tables that share our high-score.
84 std::multiset<table::Table> tables = {};
85
86 // So the worker function can give the updater some periodic updates.
87 table::Table view;
88 std::mutex lock;
89
90 // The current best table's score
91 float best = INFINITY;
92
93 // Reset
94 void reset() {
95 max = SIZE_MAX;
96 states = 0;
97 tables = {};
98 best = INFINITY;
99 }
100
101 // View helpers.
102 void update_view(const table::Table& update) {
103 std::lock_guard<std::mutex> guard(lock);
104 view = update;
105 }
106 table::Table get_view() {
107 std::lock_guard<std::mutex> guard(lock);
108 auto copy = view;
109 return copy;
110 }

113 // Score the results
114 void score_results() {
115 // Increase the states
116 ++states;
117
118 // Get the score, replace it if necessary
119 const auto score = m == metric::md ? metric::minimal_distortion(working, original, best) : metric::certainty_score(working, original, best);
120 if (best == INFINITY || score < best) {
121 best = score;
122 tables = {};
123 update_view(working);
124 }
125
126 // Add it to the list if it's good.
127 if (score == best) tables.emplace(working);
128 };

137 void anonymize_worker(const size_t& row, const size_t& col) {
138
139 // If we've run out of iterations, return what we've got
140 if (states == max) return;
141
142 // If we've run out of columns, score.
143 if (col == working.columns()) {score_results(); return;}
144
145 auto& column = working.index(col);
146
147 // Ignore non-quasi values
148 if (column.sensitivity != table::quasi) {anonymize_worker(row, col + 1); return;}
149
150 auto& cell = column.data[row];
151 const auto copy = cell;
152
153 // Get all the permutations of the cell
154 for (const auto& mut : working.mutations(cell, column, (max != -1))) {
155
156 // Return if we exhausted our states.
157 if (++states >= max) return;
158
159 // Make the change, update width so the tables prints nice.
160 cell = mut;
161
162 // We've reached the last column, move to the next
163 if (row == column.data.size() - 1) {
164
165 /*
166 * If the modification does not anonymize the rows, or
167 * we have already made more modifications than our best,
168 * we can prune.
169 *
170 * This line also illustrates how minute changes in a code base
171 * can have tremendous impact on performance, and how refined optimization can be:
172 * Without pruning, an example dataset took 2s to calculate. However, the scoring function
173 * is FASTER than k_anonymity, which means we can rely on the lazy-evaluation of boolean statements.
174 * If the centrality score does not evaluate, then C++ will not even bother k_anonymity because anything
175 * X && false = false Putting centrality first lead to a runtime of 0.2; putting k_anonymity first lead to
176 * a runtime of 0.4. In computation time, 2 tenths of a second is a massive improvement, all from the
177 * ordering of conditions in an if statement.
178 */
179 if ((m == metric::c ? metric::certainty_score(working, original, best) : metric::minimal_distortion(working, original, best) <= best) && metric::k_anonymity(working, original, k, col)) {
180 if (col == working.header.size() - 1) score_results();
181
182 // Move to the next column
183 else anonymize_worker(0, col + 1);
184 }
185 }
186
187 // Move to the next row
188 else anonymize_worker(row + 1, col);
189
190 // Once we've run through all the possible iterates with this mutation, revert the mutation.
191 cell = copy;
192 }
193 }

207 void anonymize(const size_t& k_val, const metric::metric& m_val, const size_t& iters) {
208
209 // Update
210 k = k_val;
211 m = m_val;
212 max = iters;
213 working = original;
214
215
216 // Check if we already meet k-anonymity
217 if (metric::k_anonymity(working, original, k, original.columns() - 1)) {
218 std::cout << "Already meets K-Anonymity Threshold!" << std::endl;
219 return;
220 }
221
222 // Print the metric, and iterations.
223 log("Scoring Metric: " << m);
224 auto total = working.get_distinct();
225 if (total == SIZE_MAX) {log("There are more states than your computer is capable of storing and displaying. Good luck.");}
226 else log("Possible Configurations: " << total);
227 if (max != -1) log("Non-Exhastive Search: Results may not be best");
228
229 // Pick a mode, start the timer.
230 shared::pick_mode();
231 auto start = std::chrono::steady_clock::now();
232
233 // PGO hates multi-thread, so we keep it within one thread when running for that.
234 if (shared::single_thread) anonymize_worker(0, 0);
235
236
237 else {
238
239 // Start the worker.
240 std::future<void> future = std::async(std::launch::async, &MinGen::anonymize_worker, this, 0, 0);
241
242 table::Table sample = original;
243
244 // For a smooth 60 FPS ;)
245 bool old_best = INFINITY;
246 size_t old_states = 0, speed = 0;
247 for (size_t x = 0; future.wait_for(std::chrono::milliseconds(17)) != std::future_status::ready; ++x) {
248 shared::clear();
249
250 // Print the title
251 std::cout << shared::print_title(shared::mg, x/5) << std::endl;
252
253 // Update our current best when it changes.
254 if (old_best != best) {
255 sample = get_view();
256 sample.update_widths();
257 old_best = best;
258 }
259
260 // Print some statistics.
261 std::cout << "States: " << states << "/" << total << " = ~" << speed << "/sec" << std::endl;
262 std::cout << "Note: The above upper bound does not account for pruning!" << std::endl;
263 std::cout << "Score: (Smaller is better): " << best << std::endl;
264
265 // Update roughly twice per second.
266 if (x % 30 == 0) {
267 speed = (states - old_states)*2;
268 old_states = states;
269 }
270
271 sample.print();
272 }
273 }
274
275 // Get the duration, print the results, and reset.
276 auto end = std::chrono::steady_clock::now();
277 auto duration = end - start;
278 metric::print_stats(tables, max, total, states, best, duration);
279 reset();
280 }
281 };
