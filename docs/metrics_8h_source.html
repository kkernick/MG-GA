1/*
2 * This file contains all the metrics used in the program, including
3 * k-anonymity, certainty score, and minimal_distortion.
4 *
5 * For purposes of speed, both scoring and k-anonymity are cached using
6 * The score_cache and match_cache respectively. The extent to which these increase
7 * performance, both under MinGen and GeneticAlgorithm, is so large that
8 * it was entirely unexpected. What more, the hit rate (The amount of times
9 * a value was pulled from the cache, rather than being calculated) sits
10 * at an almost perfect 1.0. Even with the caches storing everything from rows
11 * to entire tables (Implementation below) this is an exceedingly
12 * impressive rate. You can actually see the cache work as the algorithms
13 * "rev up," with the cache being populated, followed by a tremendous speedup
14 * once they start getting hit. Originally, the speed of the title strobing
15 * was tied to the amount of iterations, which made this effect more obvious
16 * but MinGen rips through generations so quickly that the effect was a little
17 * to uncomfortable to look at, so it now uses a static timing.
18 *
19 * I explicitly made a --no-cache flag to see just how pronounced the cache
20 * is. Try it!
21 *
22 * The MinGen Reference, available here:
23 * https://dataprivacylab.org/dataprivacy/projects/kanonymity/kanonymity2.pdf
24 * Specifies a unique metric based on precision, which is just minimal_distortion -1:
25 * "Precision of a generalized table is then one minus the sum of
26 * all cell distortions (normalized by the total number of cells)"
27 * This solution elected to just use minimal_distortion and certainty_score as
28 * they will be most familiar to the reader, with the former being practically
29 * identical to that outlined in the reference.
30 */

66 static float minimal_distortion(const table::Table& w, const table::Table& o, const float& best) {
67 float score = 0.0;
68
69 // Walk row by row
70 for (auto iter = w.row_begin(); iter != w.row_end(); ++iter) {
71 const auto& row = iter.get();
72
73 // If we've run into this row before, just return the cached value
74 if (shared::cache && score_cache.in(row)) score += score_cache.get(row);
75
76 // Otherwise, increment the row score for each modification.
77 else {
78 float row_score = 0.0;
79 for (size_t col = 0; col < w.columns(); ++col) {
80 const auto& original = o.index(iter.current(), col);
81 const auto& mod = row[col];
82
83 // Since we just add one, just add the weight.
84 if (original != mod) row_score += w.get_column(col).weight;
85 }
86
87 // Update the accumulator, update the cache.
88 score += row_score;
89 if (shared::cache) score_cache.add(row, row_score);
90 }
91 }
92 return score;
93 }

103 static float certainty_score(const table::Table& w, const table::Table& o, const float& best) {
104 float score = 0.0;
105
106 // Iterate row-by-row
107 for (auto iter = w.row_begin(); iter != w.row_end(); ++iter) {
108 const auto& row = iter.get();
109
110 // If cached, return it.
111 if (shared::cache && score_cache.in(row)) score += score_cache.get(row);
112
113 // Otherwise, generate the value
114 else {
115 float row_score = 0.0;
116 for (size_t col = 0; col < w.columns(); ++col) {
117
118 const auto& original = o.index(iter.current(), col);
119 const auto& mod = row[col];
120 float col_score = 0.0;
121
122 // If its the same, no penalty; if it's been entirely suppressed, +1
123 if (original == mod) continue;
124 else if (mod == "*") ++col_score;
125
126 else {
127
128 // If there's a domain graph, and our value is in it, determine
129 // The loss of information by using the more generalized
130 // name
131 const auto& column = w.get_column(col);
132 const auto& graph = column.graph;
133 if (!graph.empty() && !graph.find(mod).empty()) {
134 col_score += float(graph.breadth(mod)) / float(column.unique.size());
135 }
136
137 // If its an integer column and we have a range, determine
138 // the new bounds and compare the the original.
139 else if (column.t == table::type::integer && mod[0] == '[') {
140 col_score += float(shared::Range(std::string(mod)).range()) / float(column.range.range());
141 }
142
143 // This should never happen unless the arguments to the program are invalid,
144 // the table is corrupt or invalid, or something wrong happened during execution
145 else throw std::runtime_error("Invalid modification!");
146 }
147 col_score *= o.get_column(col).weight;
148 row_score += col_score;
149 }
150
151 // Update score and cache.
152 score += row_score;
153 if (shared::cache) score_cache.add(row, row_score);
154 }
155 }
156 return score;
157 }

167 static std::vector<size_t> match_row(const table::Table& o, const std::vector<std::string_view>& row, const size_t& c) {
168
169 // If we already have this row cached, return it.
170 // Notice we make no assumption about the original table, which means
171 // If we matched against another table it would be completely wrong.
172 if (shared::cache && match_cache.in(row, c)) return match_cache.get(row, c);
173
174 // Our matches
175 std::vector<size_t> matches = {};
176
177 // Iterate through each row in the source.
178 for (auto iter = o.row_begin(); iter != o.row_end(); ++iter) {
179 const auto& compare = iter.get();
180 bool match = true;
181
182 // Iterate through each field in the row.
183 for (size_t x = 0; x <= c; ++x) {
184
185 // If the values directly match, or the field has been suppressed, it matches.
186 if (row[x] == compare[x] || row[x] == "*") continue;
187
188 // Check if the values can be defined as a generalization from the column's graph.
189 // IE: Higher Education would match if the specialized value PhD.
190 const auto& column = o.get_column(x);
191
192 // Ignore non-quasi items.
193 if (column.sensitivity != table::quasi) continue;
194
195 // If there is a domain, find out if the value is contained within it.
196 const auto& graph = column.graph;
197 if (!graph.empty()) {
198 auto path = graph.find(row[x]);
199 if (std::count(path.begin(), path.end(), compare[x])) continue;
200 path = graph.find(compare[x]);
201 if (std::count(path.begin(), path.end(), row[x])) continue;
202 }
203
204 // If the column is integers, and we have a range, see if the value fits in it.
205 else if (column.t == table::type::integer && row[x][0] == '[') {
206 const auto range = shared::Range(std::string(row[x]));
207 if (range.in(std::atoi(compare[x].data()))) continue;
208 }
209
210 // If none applied, this value is incongrugent, and doesn't match. Break.
211 match = false;
212 break;
213 }
214
215 // If we have a match, add it.
216 if (match) matches.emplace_back(iter.current());
217 }
218
219 // Cache and return so we don't have to do this again.
220 if (shared::cache) match_cache.add(row, matches, c);
221 return matches;
222 }

251 static void k_tree(const std::vector<std::vector<size_t>>& matches, std::list<size_t>& tree, std::vector<std::set<size_t>>& ks, const size_t& x = 0) {
252
253 // If we're done recurisve search, update the ks by our
254 // path down the tree, represented one valid way to
255 // organize our matches.
256 if (x == matches.size()) {
257 size_t i = 0;
258 for (const auto& x : tree) ks[i++].emplace(x);
259 return;
260 }
261
262 // Otherwise, go through each match, see if our choices are still valid
263 // Given the choices of the matches, and emplace those options if it is.
264 for (const auto& option : matches[x]) {
265 if (std::find(tree.begin(), tree.end(), option) == tree.end()) {
266 tree.emplace_back(option);
267 k_tree(matches, tree, ks, x+1);
268 tree.pop_back();
269 }
270 }
271 }

289 static bool k_anonymity(const table::Table& w, const table::Table& o, const size_t& k, size_t c = SIZE_MAX) {
290
291 if (c == SIZE_MAX) c = w.columns() -1;
292
293 // Get all the matches for each of the modified row.
294 std::vector<std::vector<size_t>> matches;
295 for (auto iter = w.row_begin(); iter != w.row_end(); ++iter) {
296 const auto& row = iter.get();
297 auto match = match_row(o, row, c);
298
299 // Bail early if we find a row cannot even be matched here.
300 if (match.size() < k) {++match_trims; return false;}
301
302 matches.emplace_back(match);
303 }
304
305 // Use our tree algorithm to find the k score given our matches.
306 auto tree = std::vector<std::set<size_t>>(matches.size());
307 std::list<size_t> working = {};
308 k_tree(matches, working, tree, 0);
309
310 // Check to ensure each row can be matched
311 // to at least k rows in all solutions.
312 for (const auto& solutions : tree) {
313 if (solutions.size() < k) return false;
314 }
315 return true;
316 }

318 // A modified k_anonymity for GeneticAlgorithm which returns the average k anonymity
319 // of the rows, rather than a uninformative yes/no boolean.
320 static float av_k_anonymity(const table::Table& w, const table::Table& o, size_t c = SIZE_MAX) {
321
322 if (c == SIZE_MAX) c = w.columns() -1;
323 float score = 0.0;
324
325 // Get all the matches for each of the modified row.
326 std::vector<std::vector<size_t>> matches;
327 for (auto iter = w.row_begin(); iter != w.row_end(); ++iter) {
328 matches.emplace_back(match_row(o, iter.get(), c));
329 }
330
331 // Use our tree algorithm to find the k score given our matches.
332 auto tree = std::vector<std::set<size_t>>(matches.size());
333 std::list<size_t> working = {};
334 k_tree(matches, working, tree, 0);
335
336 // Check to ensure each row can be matched
337 // to at least k rows in all solutions.
338 for (const auto& solutions : tree) {
339 score += solutions.size();
340 }
341 return score / float(tree.size());
342 }
